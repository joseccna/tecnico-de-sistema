criar tabela:
CREATE TABLE Clientes (ID_cliente int NOT NULL AUTO_INCREMENT PRIMARY KEY, Nome_cliente varchar(255) not null);

create TABLE Vendedores (ID_vendedores int not null AUTO_INCREMENT PRIMARY KEY, Nome_vendedor varchar(255) not null, cpf varchar(14) UNIQUE not null, loja varchar(100));

CREATE TABLE produtos (ID_produto int not null AUTO_INCREMENT PRIMARY KEY, Nome_produto varchar(100) not null);

create table Pedidos (ID_pedidos int not null AUTO_INCREMENT PRIMARY KEY, ID_cliente int, FOREIGN KEY(ID_cliente) REFERENCES clientes(ID_cliente), ID_vendedores int, FOREIGN KEY(ID_vendedores) REFERENCES vendedores(ID_vendedores), Quantidade int not null, ID_produto int , FOREIGN KEY(ID_produto) REFERENCES produtos(ID_produto));
